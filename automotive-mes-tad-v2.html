Span<T> 实战：PLC 数据零拷贝解析
// 传统方式：每次解析分配 byte[4096] + float[4] float[] ParsePlcData(byte[] raw) { var floats = new float[4]; for (int i = 0; i < 4; i++) floats[i] = BitConverter.ToSingle(raw, i * 4); return floats; // 堆分配! } // 零分配方式：Span 零拷贝 bool TryParseOeeData(ReadOnlySpan<byte> raw, out float availability, out float performance, out float quality, out float oee) { if (raw.Length < 16) { availability = performance = quality = oee = 0; return false; } availability = BitConverter.ToSingle(raw.Slice(0, 4)); performance = BitConverter.ToSingle(raw.Slice(4, 4)); quality = BitConverter.ToSingle(raw.Slice(8, 4)); oee = BitConverter.ToSingle(raw.Slice(12, 4)); return true; // 零堆分配! }
收益测算：8 台设备 × 10 Hz 采样 = 80 次/秒解析。传统方式年分配 ≈ 2.5 亿次 float[]（~2.5 GB 堆压力）。Span 方案完全消除。
stackalloc 实战：SPC 样本统计
// SPC 均值-极差计算：每5样本一组 SpcResult ComputeXBarR(ReadOnlySpan<double> samples) { // stackalloc: 5个样本的拷贝在栈上 Span<double> sorted = stackalloc double[samples.Length]; samples.CopyTo(sorted); sorted.Sort(); // Span.Sort() - .NET 10 新增! double sum = 0; foreach (ref var v in sorted) sum += v; return new SpcResult( sum / samples.Length, // X̄ sorted[^1] - sorted[0] // R ); } // 零堆分配：sorted 在栈上，方法返回即释放

// 8台设备共享 512B 缓冲区池 public class PlcBufferPool { const int BufferSize = 512; readonly ArrayPool<byte> _pool = ArrayPool<byte>.Shared; public byte[] Rent() => _pool.Rent(BufferSize); public void Return(byte[] buffer) => _pool.Return(buffer); // 使用示例：每次 PLC 读取 public async Task<ReadOnlyMemory<byte>> ReadPlcAsync() { var buffer = _pool.Rent(BufferSize); // 从池租用 try { var len = await _socket.ReceiveAsync(buffer); return new ReadOnlyMemory<byte>(buffer, 0, len); } finally { _pool.Return(buffer); } } }

📡 ESP 产线 PLC 数据采集管道（8 设备 × 100Hz → Channel → R3 管道 → SignalR）

// ─── 架构：BoundedChannel 作为 PLC 数据缓冲 ─── public sealed class PlcDataAcquisitionPipeline : IAsyncDisposable { private readonly Channel<PlcSnapshot> _channel; private readonly CancellationTokenSource _cts = new(); public PlcDataAcquisitionPipeline(int capacity = 10000) { // BoundedChannel: 背压保护，写满时阻塞生产者 _channel = Channel.CreateBounded<PlcSnapshot>( new BoundedChannelOptions(capacity) { FullMode = BoundedChannelFullMode.Wait // 生产者等待 }); } // 生产者：8 台设备的 OPC UA 读取循环 public async Task ProduceLoopAsync() { var writer = _channel.Writer; while (!_cts.IsCancellationRequested) { foreach (var device in _devices) { var data = await ReadPlcAsync(device); // Span零拷贝读取 await writer.WriteAsync(data); // 写入 Channel } await Task.Delay(10); // 100Hz } writer.Complete(); } // 消费者：R3 管道处理数据 public IAsyncEnumerable<PlcSnapshot> ConsumeAsync() { return _channel.Reader.ReadAllAsync(_cts.Token); } }

2System.IO.Pipelines — 零拷贝网络 IO

🔌 OPC UA TCP 连接的高性能实现

// 基于 PipeReader/PipeWriter 的 OPC UA 帧读取器 public sealed class OpcUaPipeReader { private readonly PipeReader _reader; public OpcUaPipeReader(Stream stream) { _reader = PipeReader.Create(stream, new StreamPipeReaderOptions( bufferSize: 4096, minimumReadSize: 256, pool: MemoryPool<byte>.Shared // 池化内存 )); } // 零拷贝读取 OPC UA 帧 public async ValueTask<OpcUaFrame?> ReadFrameAsync() { while (true) { ReadResult result = await _reader.ReadAsync(); // ReadOnlySequence: 零拷贝多段视图 ReadOnlySequence<byte> buffer = result.Buffer; // 在 ReadOnlySequence 上直接解析，不复制 if (TryParseFrame(ref buffer, out var frame)) { _reader.AdvanceTo(buffer.Start); // 消费已解析数据 return frame; } // 数据不完整，标记检查点，等待更多数据 _reader.AdvanceTo(consumed: default, buffer.End); if (result.IsCompleted) return null; } } // 在 ReadOnlySequence 上零拷贝解析 private static bool TryParseFrame( ref ReadOnlySequence<byte> buffer, out OpcUaFrame frame) { // 使用 SequenceReader 零分配读取 var reader = new SequenceReader<byte>(buffer); if (!reader.TryReadBigEndian(out int msgType)) { frame = default; return false; } if (!reader.TryReadBigEndian(out int bodyLen)) { frame = default; return false; } if (reader.Remaining < bodyLen) { frame = default; return false; } frame = new OpcUaFrame(msgType, bodyLen); buffer = reader.UnreadSequence; // 更新剩余 buffer return true; } }
为什么不用 Stream.ReadAsync？Stream 每次读取需要分配 byte[] 缓冲区。Pipelines 使用 MemoryPool<T>.Shared 池化内存，ReadOnlySequence<T> 支持跨多个不连续缓冲区零拷贝解析。实测 OPC UA 吞吐提升 2~3x，分配减少 90%+。
